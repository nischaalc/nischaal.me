It seems I couldn't find the page you were looking for. Take a look at this cool hex clock instead!
It's pretty easy to set up too!
HTML: 
                <div id = "hex-clock"></div>

JavaScript: 
                 <script>
                    var d, h, m, s, colorVal;

                    function showTime()
                    {

                        d = new Date();
                        h = d.getHours();
                        m = d.getMinutes();
                        s = d.getSeconds();

                        if (h <= 9) h = '0' + h;
                        if (m <= 9) m = '0' + m;
                        if (s <= 9) s = '0' + s;

                        color =  '#'+h+m+s;

                        document.body.style.backgroundColor = color;

                        document.getElementById(hex-clock).innerHTML = color;

                        setTimeout(showTime, 1000);
                    }

                    showTime();

                </script>
